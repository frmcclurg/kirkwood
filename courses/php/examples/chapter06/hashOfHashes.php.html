   function PrintArray( $array, $indent )
   {
      foreach( $array as $key => $value )
      {
         $spacer = str_repeat(" &nbsp; ", $indent );

         printf( "%s Key: %s", $spacer, $key );

         if ( is_array( $value ) )
         {
            printf( "<br />" );

            // recursive call
            PrintArray( $value, $indent + 1 );
         }
         else
         {
            printf( "&nbsp; Value: %s <br />", $value );
         }
      }
   }  // PrintArray
