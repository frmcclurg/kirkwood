   <script>
      /**
       * Uses a regular expression to determine
       * if passed string is an integer.
       *
       * @param  str    Value to test.
       * @retval status Returns true if string
       *                is an integer.
       */
      function isInteger( str ) {
         var status = false;
         var regex = /^[0-9]+$/;  // one or more numbers

         if ( regex.test( str ) ||
              ( str.length == 0 ) )
         {
            status = true;
         }

         return( status );
      }

      $( 'document' ).ready( function() {
         $( 'input' ).blur( function() {
           var str = $(this).val();

            if ( ! isInteger( str ) )
            {
               alert( 'String "' + str +
                    '" is not an integer.' );
            }
         });
      });
   </script>
